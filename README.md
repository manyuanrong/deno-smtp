### Example

```ts
import { SmtpClient } from "https://deno.land/x/smtp/mod.ts"

const client = new SmtpClient()

await client.connect({
  host: "smtp.163.com",
  port: 25,
  username: "username",
  password: "password"
})

await client.send({
  from: "mailaddress@163.com",
  to: "to-address@xx.com",
  subject: "Mail Title",
  content: "Mail Contentï¼Œmaybe HTML"
})

await client.close()
```

#### TLS connection
```ts
await client.connectTLS({
  host: "smtp.163.com",
  port: 465,
  username: "username",
  password: "password"
})
```
